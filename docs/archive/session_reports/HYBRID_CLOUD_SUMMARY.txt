🚀 快速开始指南
═════════════════════════════════════════════════════════════════════════════

第 1 步: 配置本地服务 (开发环境)
────────────────────────────────────────────────────────────────
  
  # SQLite 已默认配置，无需操作
  # 如果要启用 Redis:
  sudo apt-get install redis-server
  redis-server
  
  # 或使用 Docker:
  docker run -d -p 6379:6379 redis:latest

第 2 步: 配置数据库
────────────────────────────────────────────────────────────────
  
  本地 SQLite (推荐用于开发):
    # .env 已默认配置，开箱即用
    DATABASE_TYPE=sqlite
    DATABASE_PATH=data/comic_ai.db
  
  远程 MySQL (用于生产):
    # 编辑 .env
    DATABASE_TYPE=mysql
    MYSQL_HOST=your-server.com
    MYSQL_PORT=3306
    MYSQL_USER=root
    MYSQL_PASSWORD=your-password
    MYSQL_DATABASE=comic_ai

第 3 步: 测试连接
────────────────────────────────────────────────────────────────
  
  python3 setup_database_cloud.py

第 4 步: 查看仪表板
────────────────────────────────────────────────────────────────
  
  python3 hybrid_cloud_dashboard.py

第 5 步: 在代码中使用
────────────────────────────────────────────────────────────────
  
  from src.core.database_cloud_integration import DatabaseManager
  
  # 创建管理器
  manager = DatabaseManager(db_type="sqlite")
  manager.setup_database()
  manager.setup_cache()
  
  # 使用数据库
  manager.db.execute("SELECT * FROM users")
  
  # 使用缓存
  manager.cache.set("key", "value")

📝 代码示例
═════════════════════════════════════════════════════════════════════════════

示例 1: 本地开发 (SQLite + Redis)
────────────────────────────────────────────────────────────────

from src.core.database_cloud_integration import DatabaseManager

# 初始化
manager = DatabaseManager(db_type="sqlite")
manager.setup_database(db_path="data/comic_ai.db")
manager.setup_cache(host="localhost", port=6379)

# 创建表
manager.db.execute("""
    CREATE TABLE IF NOT EXISTS portfolio (
        id INTEGER PRIMARY KEY,
        symbol TEXT,
        quantity INTEGER,
        price REAL
    )
""")

# 插入数据
manager.db.execute(
    "INSERT INTO portfolio (symbol, quantity, price) VALUES (?, ?, ?)",
    ("AAPL", 100, 150.00)
)

# 查询数据
portfolio = manager.db.fetch_all("SELECT * FROM portfolio")

# 缓存数据
manager.cache.set("portfolio", portfolio, ttl=3600)

示例 2: 远程生产 (MySQL + Redis Cloud)
────────────────────────────────────────────────────────────────

manager = DatabaseManager(db_type="mysql")
manager.setup_database(
    host="your-db-server.com",
    user="trader",
    password="password",
    database="trading_db"
)
manager.setup_cache(
    host="your-redis.redis.cache.windows.net",
    port=6380,
    password="your-password"
)

# 使用方式相同
manager.db.execute("SELECT * FROM orders")

示例 3: 混合云部署 (本地 DB + Azure 缓存)
────────────────────────────────────────────────────────────────

manager = DatabaseManager(db_type="mysql")
manager.setup_database(
    host="azure-mysql-server.mysql.database.azure.com",
    user="admin",
    password="password",
    database="comic_ai"
)
manager.setup_cache(
    host="your-redis.redis.cache.windows.net",
    port=6380,
    password="your-password"
